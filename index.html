#!/bin/bash # Tgeshka VPN — автоматическая установка сервера AmneziaWG # Запуск: curl -sSL https://raw.githubusercontent.com/ваш-аккаунт/tgeshka-vpn/main/setup.sh | sudo bash set -euo pipefail # Настройки WG_PORT=51820 WG_NETWORK="10.0.0.1/24" CLIENT_IP="10.0.0.2" CLIENT_NAME="tgeshka-client" DNS_SERVERS="1.1.1.1, 9.9.9.9" OUTPUT_DIR="/root/tgeshka_configs" RED='\033[0;31m' GREEN='\033[0;32m' YELLOW='\033[1;33m' NC='\033[0m' # Проверка root if [[ $EUID -ne 0 ]]; then echo -e "${RED}Запустите с sudo: curl -sSL <ссылка> | sudo bash${NC}" exit 1 fi # Определение интерфейса DEFAULT_IFACE=$(ip route show default | awk '/default/ {print $5; exit}') [[ -z "$DEFAULT_IFACE" ]] && DEFAULT_IFACE="eth0" echo -e "${GREEN}===== Tgeshka VPN — установка =====${NC}" # Обновление и зависимости apt update && apt upgrade -y apt install -y curl wget ufw qrencode iptables software-properties-common # Установка AmneziaWG (для Ubuntu) или WireGuard (для Debian) if grep -qi "ubuntu" /etc/os-release; then add-apt-repository ppa:amneziavpn/ppa -y apt update apt install -y amneziawg else apt install -y wireguard wireguard-tools echo -e "${YELLOW}Установлен WireGuard (на Debian AmneziaWG требует ручной сборки).${NC}" fi # Включение IP forwarding sysctl -w net.ipv4.ip_forward=1 sysctl -w net.ipv6.conf.all.forwarding=1 echo "net.ipv4.ip_forward=1" >> /etc/sysctl.conf echo "net.ipv6.conf.all.forwarding=1" >> /etc/sysctl.conf # Оптимизация sysctl (BBR, большие буферы) cat > /etc/sysctl.d/99-tgeshka.conf < server_public.key chmod 600 server_private.key SERVER_PRIV=$(cat server_private.key) SERVER_PUB=$(cat server_public.key) wg genkey | tee client_private.key | wg pubkey > client_public.key CLIENT_PRIV=$(cat client_private.key) CLIENT_PUB=$(cat client_public.key) PUBLIC_IP=$(curl -s ifconfig.me) # Конфигурация сервера cat > /etc/amneziawg/wg0.conf < "$CLIENT_CONF" <
