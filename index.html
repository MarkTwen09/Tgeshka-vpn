/* 🎰 ПОЛНОСТЬЮ ИСПРАВЛЕННЫЙ ДВИЖОК РУЛЕТКИ */ const rCanvas = document.getElementById('roulette-canvas'); const rCtx = rCanvas.getContext('2d'); function initCanvasSize() { rCanvas.width = rCanvas.parentElement.clientWidth; rCanvas.height = 75; } window.addEventListener('resize', initCanvasSize); initCanvasSize(); // Наборы предметов разной редкости const prizePool = [ { name: '0.40 SVT', type: 'SVT', amount: 0.40, rarity: 'common', color: '#1F2937' }, { name: '1.50 SVT', type: 'SVT', amount: 1.50, rarity: 'common', color: '#1F2937' }, { name: '5.00 ₽', type: 'RUB', amount: 5.00, rarity: 'common', color: '#1F2937' }, { name: '20.00 ₽', type: 'RUB', amount: 20.00, rarity: 'rare', color: '#2563EB' }, { name: '5.50 SVT', type: 'SVT', amount: 5.50, rarity: 'rare', color: '#2563EB' }, { name: '60.00 ₽', type: 'RUB', amount: 60.00, rarity: 'rare', color: '#2563EB' }, { name: '20.00 SVT', type: 'SVT', amount: 20.00, rarity: 'epic', color: '#7C3AED' }, { name: '350.00 ₽', type: 'RUB', amount: 350.00, rarity: 'epic', color: '#D97706' } ]; let rouletteItems = []; let rouletteX = 0; window.isRouletteSpinning = false; function drawLootboxScreen() { rCtx.fillStyle = '#04070C'; rCtx.fillRect(0, 0, rCanvas.width, rCanvas.height); if (rouletteItems.length === 0) { rCtx.fillStyle = '#627A95'; rCtx.font = 'bold 11px "JetBrains Mono", monospace'; rCtx.textAlign = 'center'; rCtx.fillText('РУЛЕТКА ГОТОВА К КРУЧЕНИЮ', rCanvas.width / 2, 42); return; } const itemW = 95; const itemH = 55; const gap = 8; const startY = 10; rouletteItems.forEach((item, index) => { let itemX = rouletteX + index * (itemW + gap); // Оптимизация: не рисуем то, что за экраном if (itemX + itemW < 0 || itemX > rCanvas.width) return; // Фон ячейки предмета rCtx.fillStyle = item.color; rCtx.beginPath(); rCtx.roundRect(itemX, startY, itemW, itemH, 10); rCtx.fill(); // Рамка редкости if (item.rarity === 'epic') { rCtx.strokeStyle = '#FFD700'; rCtx.lineWidth = 2; rCtx.stroke(); } else if (item.rarity === 'rare') { rCtx.strokeStyle = '#60A5FA'; rCtx.lineWidth = 1; rCtx.stroke(); } else { rCtx.strokeStyle = '#1B283A'; rCtx.lineWidth = 1; rCtx.stroke(); } // Текст награды rCtx.fillStyle = '#FFF'; rCtx.font = 'bold 12px "JetBrains Mono", monospace'; rCtx.textAlign = 'center'; rCtx.fillText(item.name, itemX + itemW / 2, startY + 33); }); } // Первичный запуск отрисовки заглушки drawLootboxScreen(); function startRouletteSpin() { if (userBalanceSVT < 5 || window.isRouletteSpinning) return; window.isRouletteSpinning = true; userBalanceSVT -= 5; updateUIBalances(); document.getElementById('open-case-btn').disabled = true; document.getElementById('open-case-btn').innerText = 'БАРАБАН КРУТИТСЯ...'; document.getElementById('win-banner').style.display = 'none'; // Алгоритм шансов дропа const roll = Math.random(); let finalPrize; if (roll < 0.65) { finalPrize = prizePool[Math.floor(Math.random() * 3)]; } else if (roll < 0.96) { finalPrize = prizePool[3 + Math.floor(Math.random() * 3)]; } else { finalPrize = prizePool[6 + Math.floor(Math.random() * 2)]; } // Заполняем ленту (генерация 45 элементов) rouletteItems = []; for (let i = 0; i < 45; i++) { if (i === 35) { rouletteItems.push(finalPrize); // 35-й элемент будет победным } else { rouletteItems.push(prizePool[Math.floor(Math.random() * prizePool.length)]); } } const itemW = 95; const gap = 8; const centerView = rCanvas.width / 2; // Начальное положение — смещаем ленту так, чтобы видеть первые элементы rouletteX = 10; // Конечная точка остановки (35-й элемент встанет ровно по центру) // Добавляем случайное смещение внутри карточки (Math.random() * 40 - 20), чтобы указатель не вставал всегда строго в центр const targetStop = -(35 * (itemW + gap) - centerView + itemW / 2) + (Math.random() * 40 - 20); let currentSpeed = 65; // Стартовая скорость скролла const minSpeed = 0.15; // Скорость, при которой пора останавливаться function animate() { // Уменьшаем скорость (имитация трения) // Чем ближе к финишу, тем медленнее едет барабан let distanceLeft = rouletteX - targetStop; if (distanceLeft <= 0) { // Если проскочили или доехали rouletteX = targetStop; drawLootboxScreen(); endRouletteSpin(finalPrize); return; } // Динамическое замедление в зависимости от оставшегося пути currentSpeed = (distanceLeft / 25) + 0.2; if (currentSpeed > 60) currentSpeed = 60; // Ограничение максималки if (currentSpeed < minSpeed) { rouletteX = targetStop; drawLootboxScreen(); endRouletteSpin(finalPrize); return; } rouletteX -= currentSpeed; // Двигаем ленту влево drawLootboxScreen(); requestAnimationFrame(animate); } // Запуск цикла анимации requestAnimationFrame(animate); } function endRouletteSpin(prize) { window.isRouletteSpinning = false; // Начисление выигрыша на баланс if (prize.type === 'SVT') { userBalanceSVT += prize.amount; } else { userBalanceRUB += prize.amount; } const banner = document.getElementById('win-banner'); banner.style.display = 'block'; if (prize.rarity === 'epic') { banner.innerText = `👑 ЭПИЧЕСКИЙ ДЖЕКПОТ! Твой дроп: +${prize.name}`; banner.style.background = 'rgba(217, 119, 6, 0.15)'; banner.style.borderColor = '#D97706'; banner.style.color = '#F59E0B'; } else if (prize.rarity === 'rare') { banner.innerText = `🔥 Отличный окуп! Выбито: +${prize.name}`; banner.style.background = 'rgba(37, 99, 235, 0.15)'; banner.style.borderColor = '#2563EB'; banner.style.color = '#60A5FA'; } else { banner.innerText = `Не повезло, ушел в минус. Дроп: ${prize.name}`; banner.style.background = 'rgba(255,255,255,0.03)'; banner.style.borderColor = 'var(--border-color)'; banner.style.color = 'var(--text-muted)'; } updateUIBalances(); updateTradingCalculations(); }
