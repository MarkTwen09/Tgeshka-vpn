#!/bin/bash # Tgeshka VPN — универсальный установщик AmneziaWG # Запуск: sudo bash tgeshka-vpn.sh set -euo pipefail # ========== НАСТРОЙКИ (при необходимости измените) ========== WG_PORT=51820 # Порт UDP WG_NETWORK="10.0.0.1/24" # Сеть для клиентов CLIENT_IP="10.0.0.2" # IP первого клиента CLIENT_NAME="tgeshka-client" # Имя клиента DNS_SERVERS="1.1.1.1, 9.9.9.9" # DNS (Cloudflare + Quad9) OUTPUT_DIR="/root/tgeshka_configs" # ============================================================ # Цвета RED='\033[0;31m' GREEN='\033[0;32m' YELLOW='\033[1;33m' NC='\033[0m' # Функция вывода заголовка header() { echo -e "${GREEN}===== Tgeshka VPN — установка =====${NC}" } # Проверка root if [[ $EUID -ne 0 ]]; then echo -e "${RED}Ошибка: этот скрипт должен выполняться от root. Используйте sudo.${NC}" exit 1 fi # Определение основного сетевого интерфейса DEFAULT_IFACE=$(ip route show default | awk '/default/ {print $5; exit}') if [[ -z "$DEFAULT_IFACE" ]]; then echo -e "${RED}Не удалось определить сетевой интерфейс. NAT может работать некорректно.${NC}" DEFAULT_IFACE="eth0" fi # Определение ОС if grep -qi "ubuntu" /etc/os-release; then OS="ubuntu" elif grep -qi "debian" /etc/os-release; then OS="debian" else echo -e "${RED}Поддерживаются только Ubuntu и Debian.${NC}" exit 1 fi header echo -e "${YELLOW}Обновление системы и установка зависимостей...${NC}" apt update && apt upgrade -y apt install -y curl wget ufw qrencode iptables software-properties-common # Установка AmneziaWG echo -e "${YELLOW}Установка AmneziaWG...${NC}" if [[ "$OS" == "ubuntu" ]]; then add-apt-repository ppa:amneziavpn/ppa -y apt update apt install -y amneziawg elif [[ "$OS" == "debian" ]]; then # Для Debian используем репозиторий WireGuard + компилируем AmneziaWG? Упростим: используем WireGuard # Но лучше предложить установку через скрипт AmneziaWG. Для надёжности поставим wireguard-tools и соберём amneziawg из исходников? # В целях простоты — установим wireguard-tools, а для обфускации предложим использовать обычный WireGuard. # Пользователь на Debian может установить AmneziaWG вручную. Ограничимся предупреждением. echo -e "${YELLOW}Debian: установка WireGuard (AmneziaWG требует ручной сборки).${NC}" apt install -y wireguard wireguard-tools fi # Включение IP forwarding echo -e "${YELLOW}Включение IP forwarding...${NC}" sysctl -w net.ipv4.ip_forward=1 sysctl -w net.ipv6.conf.all.forwarding=1 echo "net.ipv4.ip_forward=1" >> /etc/sysctl.conf echo "net.ipv6.conf.all.forwarding=1" >> /etc/sysctl.conf # Настройка sysctl (оптимизация) echo -e "${YELLOW}Применение сетевых оптимизаций ядра...${NC}" cat > /etc/sysctl.d/99-tgeshka.conf < server_public.key chmod 600 server_private.key SERVER_PRIV=$(cat server_private.key) SERVER_PUB=$(cat server_public.key) wg genkey | tee client_private.key | wg pubkey > client_public.key CLIENT_PRIV=$(cat client_private.key) CLIENT_PUB=$(cat client_public.key) # Определение публичного IP PUBLIC_IP=$(curl -s ifconfig.me) # Конфигурация сервера echo -e "${YELLOW}Создание конфигурации сервера...${NC}" cat > /etc/amneziawg/wg0.conf < "$CLIENT_CONF" <
